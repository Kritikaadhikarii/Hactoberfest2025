#!/usr/bin/env python3
"""
Requirements file for Python Scripts Collection
Install required packages for the Python scripts to work properly.
"""

# Core Python packages (usually included with Python)
# - os, sys, json, csv, sqlite3, threading, time, subprocess
# - datetime, typing, argparse

# External packages that need to be installed
REQUIRED_PACKAGES = [
    "requests>=2.25.0",           # For web scraping and API requests
    "beautifulsoup4>=4.9.0",      # For HTML parsing in web scraper
    "psutil>=5.8.0",             # For system monitoring
    "Pillow>=8.0.0",             # For image processing
]

# Optional packages for enhanced functionality
OPTIONAL_PACKAGES = [
    "pandas>=1.3.0",             # For advanced data analysis
    "numpy>=1.21.0",             # For numerical computations
    "matplotlib>=3.4.0",         # For data visualization
    "flask>=2.0.0",              # For web application development
    "fastapi>=0.68.0",           # For modern API development
    "selenium>=4.0.0",           # For advanced web scraping
    "schedule>=1.1.0",           # For advanced task scheduling
]

def install_requirements():
    """Install required packages"""
    import subprocess
    import sys
    
    print("Installing required packages...")
    
    for package in REQUIRED_PACKAGES:
        try:
            subprocess.check_call([sys.executable, "-m", "pip", "install", package])
            print(f"✅ Installed {package}")
        except subprocess.CalledProcessError:
            print(f"❌ Failed to install {package}")
    
    print("\nRequired packages installation completed!")
    print("\nTo install optional packages, run:")
    print("pip install " + " ".join(OPTIONAL_PACKAGES))

if __name__ == "__main__":
    install_requirements()
